def get_trade_fee(self, order_id: str) -> float:
    """查询指定订单的手续费

    Args:
        order_id: 订单 ID（平仓单的 orderID）

    Returns:
        float: 手续费（美元）
    """
    result = self._get(f'{API_BASE}/execution/tradeHistory', query={
        'filter': json.dumps({'orderID': order_id}),
    })

    if not isinstance(result, list) or not result:
        return 0.0

    total_fee = 0.0
    for exec_row in result:
        comm = float(exec_row.get('execComm', 0) or 0)
        total_fee += comm / 1e6
    return total_fee
