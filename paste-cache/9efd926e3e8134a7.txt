from typing import Dict, List
from core.strategy.base import Strategy, Signal, TradingSignal

class MirrorStrategy(Strategy):
    def __init__(self, inner_strategy: Strategy, interval: str = '1m'):
        super().__init__(inner_strategy.name + '_mirror')
        self._inner = inner_strategy
        self._interval = interval

    @property
    def name(self) -> str:
        return self._inner.name + '_mirror'

    @name.setter
    def name(self, value):
        pass

    def _mirror_klines(self, klines: List[Dict]) -> List[Dict]:
        """v2.1 终极翻转逻辑：2 × SMA(200) - price"""
        if not klines:
            return []
        n = min(len(klines), 200)
        sma = sum(k['close'] for k in klines[-n:]) / n
        pivot = 2 * sma

        mirrored = []
        for k in klines:
            m = dict(k)
            m['open'] = pivot - k['open']
            m['close'] = pivot - k['close']
            m['high'] = pivot - k['low']
            m['low'] = pivot - k['high']
            mirrored.append(m)
        return mirrored

    def generate_signal(self, coin: str, klines: List[Dict] = None):
        if not klines:
            return TradingSignal(signal=Signal.HOLD, coin=coin)

        # 1. 翻转K线
        flipped = self._mirror_klines(klines)

        # 2. 强制内存缓存，不污染DB
        try:
            from core.indicator_cache import init_cache
            init_cache(coin, self._interval, flipped, db_path=':memory:')
        except Exception:
            pass

        # 3. 跑策略，直接返回原信号（绝对不反转信号！）
        self._inner.current_idx = len(flipped) - 1
        return self._inner.generate_signal(coin, flipped)
