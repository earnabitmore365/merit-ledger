cur = conn.cursor()

# 假设 TTP 激活阈值 20% 盈利，trailing 5% 回撤
TTP_ACTIVATE = 0.2
TTP_TRAILING = 0.05

# 从 paired_trades 取每笔
trades = cur.execute("SELECT id, direction, entry_price, mfe FROM paired_trades")

for trade in trades:
