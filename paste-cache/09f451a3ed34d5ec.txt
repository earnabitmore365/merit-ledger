pythondef _flip_klines(self, klines):
    if not klines:
        return klines
    
    # 滚动 SMA(200)，跟 build_klines_indicators.mirror_candles 一致
    period = 200
    flipped = []
    window_sum = 0.0
    
    for i, k in enumerate(klines):
        window_sum += k['close']
        if i >= period:
            window_sum -= klines[i - period]['close']
        
        current_period = min(i + 1, period)
        sma = window_sum / current_period
        pivot = 2 * sma
        
        m = dict(k)
        m['open'] = max(1e-8, pivot - k['open'])
        m['close'] = max(1e-8, pivot - k['close'])
        m['high'] = max(1e-8, pivot - k['low'])
        m['low'] = max(1e-8, pivot - k['high'])
        flipped.append(m)
    
    return flipped
